def hitung_frekuensi(teks):
    teks = teks.replace("\n", "").replace(" ", "")  # Hapus newline dan spasi
    frekuensi = Counter(teks)

    sorted_freq = sorted(frekuensi.items(), key=lambda x: x[1], reverse=True)
    
    return sorted_freq
